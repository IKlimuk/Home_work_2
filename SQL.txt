use klimuk;

create table country (
id int PRIMARY KEY NOT NULL AUTO_INCREMENT,
name varchar (40),
area double
);

create table city (
id int PRIMARY KEY NOT NULL AUTO_INCREMENT,
name varchar (40),
number_of_population int
);
